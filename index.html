$("#locate1").click(function(){
	var url = "https://raw.githubusercontent.com/babypaunch/libparts/master/test_data.json"
	L.locate(url, {"a": 1, "b": 2}, {"bypass": {"c": 3, "d": 4}}
		, function(jsonString){
			console.log(jsonString);
		}
		, function(jsonString){
			console.log(jsonString);
			$("div[data-layered]").click(function(){
				L.ui.closeLayer(true);
			});
		}
	);
});
